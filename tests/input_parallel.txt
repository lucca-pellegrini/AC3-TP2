UNITS MULTD 2     # 2 Mul units

ADDD F1 F2 F3     # Independent: F1 = 1 + 2 = 3
ADDD F4 F5 F6     # Independent: F4 = 3 + 4 = 7 (parallel with above)
MULTD F7 F1 F4    # Depends on F1, F4: F7 = 3 * 7 = 21
MULTD F8 F2 F5    # Independent: F8 = 1 * 3 = 3 (parallel with F7)
INSTRUCTIONS_END
