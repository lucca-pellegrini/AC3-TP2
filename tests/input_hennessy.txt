# Classic Hennessy & Patterson Tomasulo example
# From "Computer Architecture: A Quantitative Approach"

CONFIG_BEGIN

CYCLES ADDD 2
CYCLES SUBD 2
CYCLES MULTD 10
CYCLES DIVD 40
CYCLES LD 2
CYCLES SD 2

UNITS ADDD 3
UNITS MULTD 2
MEM_UNITS LD 3
MEM_UNITS SD 3

CONFIG_END

REG_INIT_BEGIN
F0 0.0
F2 10.0
F4 5.0
R2 100
R3 200
REG_INIT_END

INSTRUCTIONS_BEGIN
L.D  F6  34 R2      # F6 = Mem[134]
L.D  F2  45 R3      # F2 = Mem[245]
MUL.D F0 F2 F4      # F0 = F2 * F4
SUB.D F8 F6 F2      # F8 = F6 - F2
DIV.D F10 F0 F6     # F10 = F0 / F6
ADD.D F6 F8 F2      # F6 = F8 + F2
INSTRUCTIONS_END
